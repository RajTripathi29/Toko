<script>

/* Typing name */

const nameText = "Toko";

let i = 0;

const nameSpan = document.getElementById("name");

 

function typeName() {

    if (i < nameText.length) {

        nameSpan.textContent += nameText.charAt(i);

        i++;

        setTimeout(typeName, 180);

    }

}

typeName();

 

/* Hearts */

function launchHearts(speed=700) {

    setInterval(() => {

        const heart = document.createElement("div");

        heart.className = "heart";

        heart.textContent = ["❤️","💖","💕","💘"][Math.floor(Math.random()*4)];

        heart.style.left = Math.random() * 100 + "vw";

        heart.style.animationDuration = (4 + Math.random()*2) + "s";

        document.body.appendChild(heart);

        setTimeout(() => heart.remove(), 6000);

    }, speed);

}

launchHearts(800);

 

/* Sparkles */

function launchSparkles() {

    setInterval(() => {

        const sparkle = document.createElement("div");

        sparkle.className = "sparkle";

        sparkle.textContent = "✨";

        sparkle.style.left = Math.random() * 100 + "vw";

        sparkle.style.top = Math.random() * 100 + "vh";

        document.body.appendChild(sparkle);

        setTimeout(() => sparkle.remove(), 2000);

    }, 300);

}

 

/* Burger pop */

function popBurger() {

    const burger = document.createElement("div");

    burger.textContent = "🍔";

    burger.style.position = "fixed";

    burger.style.left = "50%";

    burger.style.top = "50%";

    burger.style.fontSize = "48px";

    burger.style.transform = "translate(-50%, -50%) scale(0)";

    burger.style.transition = "transform 0.4s ease";

    document.body.appendChild(burger);

 

    requestAnimationFrame(() => {

        burger.style.transform = "translate(-50%, -50%) scale(1.4)";

    });

 

    setTimeout(() => burger.remove(), 900);

}

 

/* Buttons */

const noTexts = [

    "bubba please :(",

    "what about McD",

    "will get you burgers",

    "kisses and cuddles?"

];

 

let noClickCount = 0;

const yesBtn = document.getElementById("yesBtn");

const noBtn = document.getElementById("noBtn");

 

noBtn.addEventListener("click", () => {

 

    if (noClickCount < noTexts.length) {

        noBtn.textContent = noTexts[noClickCount];

    }

 

    noClickCount++;

 

    if (noClickCount === 2) popBurger();

 

    yesBtn.style.animation = "none";

    yesBtn.style.transform = `scale(${1 + noClickCount * 0.35})`;

 

    if (noClickCount >= 4) {

        noBtn.textContent = "Yes";

        noBtn.style.background = "#ff4d6d";

        noBtn.onclick = celebrate;

    }

});

 

yesBtn.addEventListener("click", () => {

    yesBtn.style.animation = "none";

    celebrate();

});

 

/* Celebration */

function celebrate() {

    if (navigator.vibrate) navigator.vibrate([200,100,200]);

 

    document.getElementById("proposalBox").style.display = "none";

    document.getElementById("celebration").style.display = "block";

    document.getElementById("glowLayer").style.opacity = "1";

 

    const music = document.getElementById("bgMusic");

    music.volume = 0.5;

    music.play().catch(()=>{});

 

    launchHearts(250);

    launchSparkles();

}
